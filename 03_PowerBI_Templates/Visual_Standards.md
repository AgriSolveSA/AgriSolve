// Pseudocode: Dashboard Visual Design Rules

DEFINE VisualDesignConstraints:

    Rule_MaxVisualsPerPage:
        - Each dashboard page must contain <= 6 visuals

    Rule_ColorScheme:
        - Use neutral colors only
        - Avoid bright, distracting palettes
        - Ensure readability and consistency across pages

APPLY Constraints:
    - Enforce Rule_MaxVisualsPerPage for all defined pages
    - Apply Rule_ColorScheme to all visuals
